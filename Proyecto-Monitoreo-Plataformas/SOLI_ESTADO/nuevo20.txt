        access_log /var/log/invoice-app/frontend/access.log;
        error_log /var/log/invoice-app/frontend/error.log;

    # Backend API - proxy inverso
    location /api/ {
        proxy_pass http://localhost:5000/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection keep-alive;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_cache_bypass $http_upgrade;

        # Logs
        access_log /var/log/invoice-app/api/access.log;
        error_log /var/log/invoice-app/api/error.log;

        deny all;
    }

    # Configuración de Gzip para mejorar el rendimiento

    gzip_types text/plain text/css application/json application/javascript text/xml application/xml application/xml+rss text/javascript;
